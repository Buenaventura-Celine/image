BitmapFont.fromFnt(String fnt, Image page) {
  final fontPages = {0: page};

  XmlDocument doc;
  fnt = fnt.trimLeft();

  if (fnt.startsWith('<?xml') || fnt.startsWith('<font>')) {
    doc = XmlDocument.parse(fnt);
  } else {
    doc = _parseTextFnt(fnt);
  }

  _parseFnt(doc, fontPages);
}
